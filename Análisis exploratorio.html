import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# Asumiendo que tu DataFrame se llama df

# Filtrar los registros donde AGOTADO es 0 (No Agotados)
no_agotados = df[df['AGOTADO'] == 0]

# Ver el n√∫mero de registros de no agotados
print(f'N√∫mero de registros de No Agotados: {len(no_agotados)}')

# Opcional: Ver un resumen de las primeras filas de los no agotados
print(no_agotados.head())

# Agrupar por 'CATEGORIA' y 'REGION' y contar los registros de no agotados
region_category_counts = no_agotados.groupby(['CATEGORIA', 'REGION']).size().reset_index(name='Count')

# Pivotar los datos para obtener una estructura adecuada para las barras apiladas
pivot_data = region_category_counts.pivot_table(index='CATEGORIA', columns='REGION', values='Count', fill_value=0)

# Ordenar las categor√≠as por la cantidad total de productos no agotados (de menor a mayor)
pivot_data['Total'] = pivot_data.sum(axis=1)
pivot_data = pivot_data.sort_values(by='Total', ascending=True)  # Cambi√© a ascending=True

# Calcular los porcentajes
pivot_data_percent = pivot_data.drop(columns='Total').div(pivot_data['Total'], axis=0) * 100

# Crear la visualizaci√≥n de barras apiladas horizontales con porcentajes
plt.figure(figsize=(12, 8))
